\section{Verse Calculus Is Skew-Confluent}


\subsection{Modified Grammar and Rewrite Rules}

Let us modify the grammar for \versecalc{} to have one additional kind of value, a~\emph{fixpoint} value \ensuremath{\mu\Varid{x}.\,\mathit{hnf}}:
\begin{quote}
$
\begin{array}{lr@{\;}c@{\;}l}
      \text{Values} & \ensuremath{\Varid{v}} & ::= & \ensuremath{\Varid{x}}  \vb \hnf \vb \ensuremath{\mu\Varid{x}.\,\mathit{hnf}} \\
\end{array}
$
\end{quote}
and a modify the set of Unification rewrite rules $\runi$ so that rule \rulename{u-occurs}
\begin{quote}
$
\begin{array}{@{\quad}lr@{\;}c@{\;} l l}
  \rulename{u-occurs} & \ensuremath{\Varid{x}\mathrel{=}\Conid{V} [\mskip1.5mu \Varid{x}\mskip1.5mu]} & \movesto & \ensuremath{\textbf{fail}} & \text{if $\ensuremath{\Conid{V}} \ne \ensuremath{\hole}$} \\
\end{array}
$
\end{quote}
is replaced by the two rules\footnote{These two rules allow equations to be recursive through lambda expressions and possibly also tuples, but not through tuples only;
  thus equations such as \ensuremath{\Varid{f}\mathrel{=}\lambda\Varid{y}.\,\langle{}\Varid{y},\Varid{f}\rangle} and \ensuremath{\Varid{x}\mathrel{=}\langle{}\mathrm{1},\lambda\Varid{y}.\,\langle{}\Varid{y},\Varid{x}\rangle\rangle} can be processed by rule \rulename{u-occurs-wrap},
  but the equation \ensuremath{\Varid{x}\mathrel{=}\langle{}\mathrm{1},\Varid{x}\rangle} can be processed only by rule \rulename{u-occurs-fail}.
  An alternate design using the single rule
  \begin{quote}
$
\begin{array}{@{\quad}lr@{\;}c@{\;} l l}
  \rulename{u-occurs-wrap} & \ensuremath{\Varid{x}\mathrel{=}\mathit{hnf}} & \movesto & \ensuremath{\Varid{x}\mathrel{=}\mu\Varid{x}.\,\mathit{hnf}} & \text{if $\ensuremath{\Varid{x}} \in \freevars{\ensuremath{\mathit{hnf}}}$} \\
\end{array}
$
\end{quote}
plus rule \rulename{u-unwrap} could be used instead to support recursion through tuples only as well as through lambda expressions.}\footnote{
  \rulename{u-occurs-fail} is identical to \rulename{u-occurs} in its effect,
  but it is re-expressed using $\freevarsoutsidelambdas{\cdot}$, which we need anyway for \rulename{u-occurs-wrap}.  Now we can drop the context $V$, which was only
  used in \rulename{u-occurs}.
}
\begin{quote}
$
\begin{array}{@{\quad}lr@{\;}c@{\;} l l}
  \rulename{u-occurs-fail} & \ensuremath{\Varid{x}\mathrel{=}\mathit{hnf};\,\Varid{e}} & \movesto & \ensuremath{\textbf{fail}} & \text{if $\ensuremath{\Varid{x}} \in \freevarsoutsidelambdas{\ensuremath{\mathit{hnf}}}$} \\
  \rulename{u-occurs-wrap} & \ensuremath{\Varid{x}\mathrel{=}\mathit{hnf};\,\Varid{e}} & \movesto & \ensuremath{\Varid{x}\mathrel{=}\mu\Varid{x}.\,\mathit{hnf};\,\Varid{e}} & \text{if $\ensuremath{\Varid{x}} \in \freevars{\ensuremath{\mathit{hnf}}}$ and $\ensuremath{\Varid{x}} \not\in \freevarsoutsidelambdas{\ensuremath{\mathit{hnf}}}$} \\
\end{array}
$
\end{quote}
Let us also add this rewrite rule:
\begin{quote}
$
\begin{array}{@{\quad}lr@{\;}c@{\;} l l}
  \rulename{u-unwrap} & \ensuremath{\mu\Varid{x}.\,\mathit{hnf}} & \movesto & \ensuremath{\subst{\mathit{hnf}}{\Varid{x}}{\mu\Varid{x}.\,\mathit{hnf}}} \\
\end{array}
$
\end{quote}
As we will see, rule \rulename{u-unwrap} is confluent but not Noetherian.

The resulting grammar is not confluent; in particular, it suffers from the even-odd problem
described in \cref{sec:even-odd} \cite[Example 4.1]{ariola:skew-confluence}.
Therefore we will modify the proof of confluence for $\runi$ to become a proof of \emph{skew confluence} \cite{ariola:skew-confluence},
and then go on to prove that \versecalc{} itself, with this modification, is skew confluent.

\subsubsection{Unwrapping of Fixpoints is Confluent but not Noetherian}

\begin{lemma}\label{lem:unwrap-confluent}
The rule \rulename{u-unwrap} is confluent.
\end{lemma}

\begin{proof}
  Suppose that $\rstep{\rulename{u-unwrap}}{\ensuremath{\Varid{e}}}{\ensuremath{\Varid{e}_{\mathrm{1}}}}$ and $\rstep{\rulename{u-unwrap}}{\ensuremath{\Varid{e}}}{\ensuremath{\Varid{e}_{\mathrm{2}}}}$ for distinct redexes within $e$.
  
  If the redexes are disjoint, then \cref{lem:disjoint} applies.

  Otherwise, without loss of generality assume the redex for $\rstep{\rulename{u-unwrap}}{\ensuremath{\Varid{e}}}{\ensuremath{\Varid{e}_{\mathrm{1}}}}$ contains the redex for $\rstep{\rulename{u-unwrap}}{\ensuremath{\Varid{e}}}{\ensuremath{\Varid{e}_{\mathrm{2}}}}$;
  let $e$ must be of the form \ensuremath{\Conid{C}_{\mathrm{1}} [\mskip1.5mu \mu\Varid{x}.\,\Conid{C}_{\mathrm{2}} [\mskip1.5mu \mu\Varid{y}.\,\mathit{hnf}\mskip1.5mu]\mskip1.5mu]} ($\alpha$-conversion may be used to ensure that $x$ and $y$ are distinct variables).

  Then \ensuremath{\Varid{e}_{\mathrm{1}}\mathrel{=}\Conid{C}_{\mathrm{1}} [\mskip1.5mu \subst{\Conid{C}_{\mathrm{2}} [\mskip1.5mu \mu\Varid{y}.\,\mathit{hnf}\mskip1.5mu]}{\Varid{x}}{\mu\Varid{x}.\,\Conid{C}_{\mathrm{2}} [\mskip1.5mu \mu\Varid{y}.\,\mathit{hnf}\mskip1.5mu]}\mskip1.5mu]}
  and \ensuremath{\Varid{e}_{\mathrm{2}}\mathrel{=}\Conid{C}_{\mathrm{1}} [\mskip1.5mu \mu\Varid{x}.\,\Conid{C}_{\mathrm{2}} [\mskip1.5mu \subst{\mathit{hnf}}{\Varid{y}}{\mu\Varid{y}.\,\mathit{hnf}}\mskip1.5mu]\mskip1.5mu]}.

  From \ensuremath{\Varid{e}_{\mathrm{2}}} we can take just one more \rulename{u-unwrap} step, using the outermost redex \ensuremath{\mu\Varid{x}.\,\Conid{C}_{\mathrm{2}} [\mskip1.5mu \Varid{cdots}\mskip1.5mu]}, obtaining \ensuremath{\Varid{e'}\mathrel{=}\Conid{C}_{\mathrm{1}} [\mskip1.5mu \subst{(\Conid{C}_{\mathrm{2}} [\mskip1.5mu \subst{\mathit{hnf}}{\Varid{y}}{\mu\Varid{y}.\,\mathit{hnf}}\mskip1.5mu])}{\Varid{x}}{\mu\Varid{x}.\,\Conid{C}_{\mathrm{2}} [\mskip1.5mu \subst{\mathit{hnf}}{\Varid{y}}{\mu\Varid{y}.\,\mathit{hnf}}\mskip1.5mu]}\mskip1.5mu]}.

  \skewtodo{More to come.}

\subsubsection{Information Content}

We define a second grammar, for a second language $\vcOmega$, by adding one more value \ensuremath{\mathit{\Omega}}, which indicates a lack of information as to just what will be computed.
When \ensuremath{\mathit{\Omega}} appears in a context where only a value is permitted, it indicates uncertainty as to what value will be provided; 
when \ensuremath{\mathit{\Omega}} appears in a context where any expression is permitted, it furthermore indicates uncertainty as to how many values will be provided (possibly none).
\begin{quote}
$
\begin{array}{lr@{\;}c@{\;}l}
      \text{Values} & \ensuremath{\Varid{v}} & ::= & \ensuremath{\Varid{x}} \vb \hnf \vb \ensuremath{\mu\Varid{x}.\,\mathit{hnf}} \vb \ensuremath{\mathit{\Omega}} \\
\end{array}
$
\end{quote}

Moreover, we add a new kind of expression \ensuremath{\mathbin{?}\Varid{e}}, which explicitly indicates a lack of information as to whether \ensuremath{\Varid{e}} might be replaced by \ensuremath{\textbf{fail}} because of the failure of an
expression or equation in some execution context containing \ensuremath{\Varid{e}}; it furthermore indicates uncertainty as to whether any of its free variables might be unified with some
value.

\begin{quote}
$
\begin{array}{lr@{\;}c@{\;}l}

\end{quote}


For every term of \versecalc{} there is a corresponding term of $\vcOmega$, identical in structure and appearance and containing no occurrence of \ensuremath{\mathit{\Omega}} or \ensuremath{\mathbin{?}};

The definition of substitution (\cref{fig:substitution}) is extended in the expected trivial manner: $\ensuremath{\subst{\mathit{\Omega}}{\Varid{x}}{\Varid{v}}} \equiv \ensuremath{\mathit{\Omega}}$ and $\ensuremath{\subst{(\mathbin{?}\Varid{e})}{\Varid{x}}{\Varid{v}}} \equiv \ensuremath{\mathbin{?}(\subst{\Varid{e}}{\Varid{x}}{\Varid{v}})}$.

This primitive relation is a quasiorder but not a partial order; for example,  $\ensuremath{\mathit{\Omega}} \urleqomegaof{T} \ensuremath{\mathbin{?}\mathit{\Omega}}$ and $\ensuremath{\mathbin{?}\mathit{\Omega}} \urleqomegaof{T} \ensuremath{\mathit{\Omega}}$,

\begin{lemma}\label{lem:critpairs}\cite[Lemma 3.1]{Huet80}
The relation $\go{\rr}$ is locally confluent iff for every critical pair \ensuremath{(\Varid{e}_{\mathrm{1}},\Varid{e}_{\mathrm{2}})} of $\rr$, \ensuremath{\Varid{e}_{\mathrm{1}}} and \ensuremath{\Varid{e}_{\mathrm{2}}} can be \emph{joined}---that is, there exists \ensuremath{\Varid{e}_{\mathrm{3}}} such that
 $\ensuremath{\Varid{e}_{\mathrm{1}}} \gos{\rr} \ensuremath{\Varid{e}_{\mathrm{3}}}$ and  $\ensuremath{\Varid{e}_{\mathrm{2}}} \gos{\rr} \ensuremath{\Varid{e}_{\mathrm{3}}}$.
\end{lemma}

Observe that if the two terms of a critical pair \ensuremath{(\Varid{e}_{\mathrm{1}},\Varid{e}_{\mathrm{2}})} are identical, they can be joined at $\ensuremath{\Varid{e}_{\mathrm{3}}} = \ensuremath{\Varid{e}_{\mathrm{1}}} = \ensuremath{\Varid{e}_{\mathrm{2}}}$.
We say that such a critical pair is \emph{trivial}, and no separate proof of joining is required.


\begin{lemma}[$\rinfo$-Confluent] \label{lem:rinfo-c}
$\rinfo$ is locally confluent and Noetherian; therefore $\rinfo$ is confluent.
\end{lemma}

\begin{proof}
Consider all critical pairs of $\rinfo$. By inspection, we can easily see that critical pairs can arise only from left-hand-side subterms of the form \ensuremath{\mathbin{?}\Varid{e}} or \ensuremath{\Varid{e}\;\rule[-0.1em]{0.15em}{0.75em}\;\Varid{e'}}, so we examine these possibilities in more detail.
\begin{itemize}
\item  Rules \rulename{info-unsure-fail} and \rulename{info-unsure-omega} produce no critical pairs.

\item  Rules \rulename{info-unsure-fail} and \rulename{info-unsure-unsure} produce the trivial critical pair \ensuremath{(\mathbin{?}\textbf{fail},\mathbin{?}\textbf{fail})}. 

\item  Rules \rulename{info-unsure-omega} and \rulename{info-unsure-unsure} produce the trivial critical pair \ensuremath{(\mathbin{?}\mathit{\Omega},\mathbin{?}\mathit{\Omega})}. 

\item  Rules \rulename{info-fail-l} and \rulename{info-fail-r} produce the trivial critical pair \ensuremath{(\textbf{fail},\textbf{fail})}. 

\item  Rules \rulename{info-fail-l} and \rulename{info-choice-omega} produce no critical pairs.

\item  Rules \rulename{info-fail-l} and \rulename{info-choice-assoc} produce the critical pair \ensuremath{(\Varid{e}_{\mathrm{2}}\;\rule[-0.1em]{0.15em}{0.75em}\;\Varid{e}_{\mathrm{3}},\textbf{fail}\;\rule[-0.1em]{0.15em}{0.75em}\;(\Varid{e}_{\mathrm{2}}\;\rule[-0.1em]{0.15em}{0.75em}\;\Varid{e}_{\mathrm{3}}))}.

\item  Rules \rulename{info-fail-r} and \rulename{info-choice-omega} produce the trivial critical pair \ensuremath{(\mathit{\Omega},\mathit{\Omega})}.

\item  Rules \rulename{info-fail-r} and \rulename{info-choice-assoc} produce the critical pairs \ensuremath{(\Varid{e}_{\mathrm{1}}\;\rule[-0.1em]{0.15em}{0.75em}\;\Varid{e}_{\mathrm{3}},\Varid{e}_{\mathrm{1}}\;\rule[-0.1em]{0.15em}{0.75em}\;(\textbf{fail}\;\rule[-0.1em]{0.15em}{0.75em}\;\Varid{e}_{\mathrm{3}}))} and \ensuremath{(\Varid{e}_{\mathrm{1}}\;\rule[-0.1em]{0.15em}{0.75em}\;\Varid{e}_{\mathrm{2}},\Varid{e}_{\mathrm{1}}\;\rule[-0.1em]{0.15em}{0.75em}\;(\Varid{e}_{\mathrm{2}}\;\rule[-0.1em]{0.15em}{0.75em}\;\textbf{fail}))}.

\item  Rules \rulename{info-choice-omega} and \rulename{info-choice-assoc} produce the critical pair \ensuremath{(\mathit{\Omega}\;\rule[-0.1em]{0.15em}{0.75em}\;\Varid{e}_{\mathrm{3}},\mathit{\Omega}\;\rule[-0.1em]{0.15em}{0.75em}\;(\Varid{e}_{\mathrm{2}}\;\rule[-0.1em]{0.15em}{0.75em}\;\Varid{e}_{\mathrm{3}})}.

\item  No other pairs of rules produce any critical pairs.
\end{itemize}
The critical pair \ensuremath{(\Varid{e}_{\mathrm{2}}\;\rule[-0.1em]{0.15em}{0.75em}\;\Varid{e}_{\mathrm{3}},\textbf{fail}\;\rule[-0.1em]{0.15em}{0.75em}\;(\Varid{e}_{\mathrm{2}}\;\rule[-0.1em]{0.15em}{0.75em}\;\Varid{e}_{\mathrm{3}}))} can be joined at \ensuremath{\Varid{e}_{\mathrm{2}}\;\rule[-0.1em]{0.15em}{0.75em}\;\Varid{e}_{\mathrm{3}}} in one step by using rule \rulename{info-fail-l} on \ensuremath{\textbf{fail}\;\rule[-0.1em]{0.15em}{0.75em}\;(\Varid{e}_{\mathrm{2}}\;\rule[-0.1em]{0.15em}{0.75em}\;\Varid{e}_{\mathrm{3}}))}.

The critical pair \ensuremath{(\Varid{e}_{\mathrm{1}}\;\rule[-0.1em]{0.15em}{0.75em}\;\Varid{e}_{\mathrm{2}},\Varid{e}_{\mathrm{1}}\;\rule[-0.1em]{0.15em}{0.75em}\;(\Varid{e}_{\mathrm{2}}\;\rule[-0.1em]{0.15em}{0.75em}\;\textbf{fail}))} can be joined at \ensuremath{\Varid{e}_{\mathrm{1}}\;\rule[-0.1em]{0.15em}{0.75em}\;\Varid{e}_{\mathrm{2}}} in one step by using rule \rulename{info-fail-r} on (a subterm of) \ensuremath{\Varid{e}_{\mathrm{1}}\;\rule[-0.1em]{0.15em}{0.75em}\;(\Varid{e}_{\mathrm{2}}\;\rule[-0.1em]{0.15em}{0.75em}\;\textbf{fail}))}. 
The critical pair \ensuremath{(\mathit{\Omega}\;\rule[-0.1em]{0.15em}{0.75em}\;\Varid{e}_{\mathrm{3}},\mathit{\Omega}\;\rule[-0.1em]{0.15em}{0.75em}\;(\Varid{e}_{\mathrm{2}}\;\rule[-0.1em]{0.15em}{0.75em}\;\Varid{e}_{\mathrm{3}})} can be joined at \ensuremath{\mathit{\Omega}} by using rule \rulename{info-choice-omega} on \ensuremath{\mathit{\Omega}\;\rule[-0.1em]{0.15em}{0.75em}\;\Varid{e}_{\mathrm{3}}} and on \ensuremath{\mathit{\Omega}\;\rule[-0.1em]{0.15em}{0.75em}\;(\Varid{e}_{\mathrm{2}}\;\rule[-0.1em]{0.15em}{0.75em}\;\Varid{e}_{\mathrm{3}})}.
Trivial critical pairs require no separate proof of joining.

All critical pairs can be joined; therefore by \cref{lem:critpairs} $\rinfo$ is locally confluent.

Let the \emph{size} of a term of $\vcOmega$ be the sum of (i)~the number of tokens it contains other than parentheses,
and (ii)~the sum, over every subterm of the term that is a choice expression, the size of the left-hand operand of that choice expression.
Because any given term $e$ has some finite number $n$ of tokens other than parentheses and a finite number $k$ of subterms that are choice expressions,
and because the size of a subterm cannot be greater than the size of an expression, the size of $e$ cannot exceed $(k+1)n$ and therefore is finite.
Furthermore the size of a term cannot be less than zero.
It follows that every rewriting sequence from term $e$ can have no more than $(k+1)n$ steps. So $\rinfo$ is bounded and therefore Noetherian.
Each of the rewrite rules in \cref{fig:info-rewrite-rules} produces a new expression that is strictly smaller in size. 

Because $\rinfo$ is locally confluent and Noetherian, it is confluent by Newman's lemma.
\end{proof}

\subsection {Proof that \texorpdfstring{\versecalc{}}{VC} Is Skew Confluent}


Our goal is to show that for all rules \rulename{r} and \rulename{s} (possibly the same), for all terms $e$, $e_1$, and $e_2$, and for all sets of occurrences $U$ and occurrences $w$,
if   $e_1 \xrnungosup{r}{U} e \xrngosup{s}{w} e_2$
then
  $e_1 \leqomegavc \ejoin \xrnungosup{r}{u} e_2$
  for some term $\ejoin$.

Our overall strategy is to use an expanded rule set that includes a rule that is the inverse of \rulename{subst};
show that for all rules \rulename{r} and \rulename{s} (possibly the same), for all terms $e$, $e_1$, and $e_2$, and for all sets of occurrences $U$ and occurrences $w$,
if   $e_1 \xrnungosup{r}{u} e \xrngosup{s}{w} e_2$
then
  $e_1 \xrnbothdblsup{}{\preceq} \ejoin \xrnungosup{r}{u} e_2$
  for some term $\ejoin$.

  Our specific strategy is to use a tiling strategy based on tiles constructed so that
for all rules \rulename{r} and \rulename{s} (possibly the same), for all terms $e$, $e_1$, and $e_2$, and for all sets of occurrences $U$ and occurrences $w$,
if   $e_1 \xprnungosup{r}{U} e \xrngosup{s}{w} e_2$
then
  $e_1 \xprnbothdblsup{}{\preceq} \ejoin \xprnungosup{r}{U} e_2$
  for some term $\ejoin$.

  Fortunately, tiles of that form are easy to construct if \rulename{r} and \rulename{s} strongly commute, and
we already know that all rules in $\rapp \cup \rgc \cup \rspc$ have the diamond property and that all pairs of rules in $\rapp \cup \rgc \cup \rspc$ strongly commute.
We need only prove separately the ``difficult'' cases.


\subsubsection{Preliminary Lemmas}

\begin{lemma}\label{lem:disjoint-sc}
  For any two rules \rulename{r} and \rulename{s}, and for all terms $e$, $e_1$, and $e_2$:\hfill\break\null\hskip3.5em if\quad
  $e_1 \xrnungosup{r}{u} e \xrngosup{s}{w} e_2$
  \unskip\quad and\quad
  $u \occdis w$
  \unskip\quad then\quad
  $e_1 \xrngosup{s}{w} \ejoin \xrnungosup{r}{u} e_2$
  \hfill\break for some term $\ejoin$.
\end{lemma}
\begin{proof}
  PROBLEM
  From $e \xrngosup{r}{u} e_1 \xrngosup{s}{w} e'$ and $u \occdis w$ we have $e' = \tsubst{\tsubst{e}{u}{e_1/u}}{w}{e'/w}$,
  from $e \xrngosup{s}{w} e_2 \xrngosup{r}{u} e''$ and $u \occdis w$ we have $e'' = \tsubst{\tsubst{e}{w}{e_2/w}}{u}{e''/u}$,
    and we have $e' = e''$ by \emph{commutativity}; therefore let $\ejoin = e_1 = e_2$.
\end{proof}

\begin{lemma}\label{lem:set-disjoint-sc}
  For any two rules \rulename{r} and \rulename{s}, and for all terms $e$, $e_1$, and $e_2$:\hfill\break\null\hskip3.5em if\quad
  $e_1 \xprnungosup{r}{U} e \xrngosup{s}{w} e_2$
  \unskip\quad and\quad
  $u \occdis w$
  \unskip\quad then\quad
  $e_1 \xrngosup{s}{w} \ejoin \xprnungosup{r}{U} e_2$
  \hfill\break for some term $\ejoin$.
\end{lemma}
\begin{proof}
  PROBLEM
  From $e \xrngosup{r}{u} e_1 \xrngosup{s}{w} e'$ and $u \occdis w$ we have $e' = \tsubst{\tsubst{\tsubst{\tsubst{e}{u_1}{e_1/u_1}}{u_2}{e_1/u_2}\cdots}{u_n}{e_1/u_n}}{w}{e'/w}$
  from $e \xrngosup{s}{w} e_2 \xrngosup{r}{U} e''$ and $u \occdis w$ we have $e'' = \tsubst{\tsubst{e}{w}{e_2/w}}{u}{e_1/u}$,
    and we have $e' = e''$ by repeated application of \emph{commutativity}; therefore let $\ejoin = e_1 = e_2$.
\end{proof}

\begin{lemma}\label{lem:diamond-par-sc}
If $\xrngo{r}$ strongly commutes with itself, then $\xprngo{r}$ strongly commutes with $\xrngo{r}$.
\end{lemma}

\begin{lemma}\label{lem:sc-par-sc}
If $\xrngo{r}$ strongly commutes with $\xrngo{s}$, then $\xprngo{r}$ strongly commutes with $\xrngo{s}$.
\end{lemma}

\begin{lemma}[\rulename{u-occurs-fail} strongly commutes with itself]\label{lem:u-occurs-fail-diamond}
  For all terms $t$,
  $t_1$, and $t_2$:\hfill\break\null\hskip2.5em if\quad
  $t_1 \xrnungosup{u-occurs-fail}{u} t \xrngosup{u-occurs-fail}{w} t_2$
  \unskip\quad then\quad
  $t_1 \xrngoeqsup{u-occurs-fail}{w} \tjoin \xrnungoeqsup{u-occurs-fail}{u} t_2$
  \hfill\break for some term $\tjoin$.
\end{lemma}
\begin{proof}
  Split cases on the position of $u$ relative to $w$ within $t$:
  \begin{description}
    \CASE{u=w} Let $\tjoin = t_1 = t_2$.
    \CASE{u \occdis w} Use \cref{lem:disjoint-sc}.
    \CASE{u \occstrictin w} Define $\tjoin$ by $t_1 \xrngosup{u-occurs-fail}{w} \tjoin$, then observe that\hfill\break

    \CASE{u \occstrictcontains w} Symmetrical with case $u \occstrictin w$.
  \end{description}
\end{proof}
