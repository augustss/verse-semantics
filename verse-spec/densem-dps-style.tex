  $$
  \begin{array}{l} 

\textbf{Domains (sets)}                              \\
\begin{array}{rcll}
  \Val   & = &  \Z + \seq{\parfun\Val\Val}           \\
  \Env   & = & \parfun \Ident \Val                   \\
\end{array}                                       \\ \\


\begin{array}{rcl}
 \E \op {}                  & : & \totfun {\Set\Ident} {\seq{\Set\Env}}                         \\
 \E \Noop \X                & = & \seq {\Envs\X}                                                \\
 \E {\exi\x} \X             & = & \seq{\Envs\X}                                                 \\
 \E {\x=\k} \X              & = & \seq{ \setcomp \rho {\pick\rho{\Envs\X}, \rho(\x)=\k }}       \\
 \E {\x=\y} \X              & = & \seq{ \setcomp \rho {\pick\rho{\Envs\X}, \rho(\x)=\rho(\y) }} \\
 \E {\x=\Lam} \X            & = & \seq{ \setcomp \rho {\pick\rho{\Envs\X},
                                                       \rho(\x)\in \F\Lam }}                    \\
 \E {\x=\y[\z]} \X          & = & \bigcup_{\fs\in\seq{\parfun\Val\Val}}
                                \seqcomp{ \setcomp \rho { \pick\rho{\Envs\X},
                                                         \rho(\y)=\fs,
                                                         \rho(\x)=\f(\rho(\z))
                                                        }}
                                        {\pick \f \fs}                                         \\
 \E {\op ; \opp} \X         & = & \seqcomp { \s \cap \sp }
                                          { \pick\s{\E\op\X}, \pick\sp{\E\opp\X}}              \\
 \E {\op | \opp} \X         & = & \E\op\X ++ \E\opp\X                                          \\
  \E \fail \X                & = & \seq{}                                                       \\
 \E {\scope\op}\X           & = & \seqcomp {\s-\Y } { \pick \s{\E\op{(\X\cup\Y)}}}
                                 \where  {\Y = \Exis\op}                                     
\end{array}

\\ \\

\F{}{} \text{returns a set of functions. For closed lambdas that set has at most one element.}

\\

\begin{array}{rcl}
 \F {\lamany \x\op\opp} {}       & : & \Set {\seq{\parfun\Val\Val}}                                   \\
\end{array}



  \end{array}
