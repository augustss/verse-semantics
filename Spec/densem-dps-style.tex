\begin{figure}
  $$
  \begin{array}{l}

\textbf{Domains (sets)}                              \\
\begin{array}{rcll}
  \Val   & = &  \Z + \Fun                            \\
  \Fun   & = & \seq{\parfun\Val\Val}                 \\
  \Env   & = & \parfun \Ident \Val                   \\
  \Ws    & = &  \seq{\Set\Env}                       \\
\end{array}                                       \\ \\

\\
\textbf{Sematics}                                    \\
\begin{array}{rcl}
 \E \op {}                  & : & \totfun {\Set\Ident} \Ws                                      \\
 \E {\exi\x} \X             & = & \seq{\Envs\X}                                                 \\
 \E {\x=\k} \X              & = & \seq{ \setcomp \rho {\pick\rho{\Envs\X}, \inrho\x=\k }}       \\
 \E {\x=\y} \X              & = & \seq{ \setcomp \rho {\pick\rho{\Envs\X}, \inrho\x=\inrho\y }} \\
 \E {\x=\Lam} \X            & = & \seq{ \setcomp \rho {\pick\rho{\Envs\X},
                                                       \rho(\x)\in \F\Lam\rho }}                \\
 \E {\x=\y[\z]} \X          & = &
                     \begin{array}[t]{l@{~}l}
                     \UU_{\fs\in\Fun} [\SP\{ \rho \SP\alt & \pick\rho{\Envs\X} \and  \\
                                                         & \inrho\y=\fs \and
                                                          \pairin{\inrho\z}{\inrho\x}\f \SPP\}
                                                         \SPP\alt\SPP {\pick \f \fs} \SPP ]
                                        \end{array} \\
 \E {\op \nxt \opp} \X      & = & \seqcomp { \s \cap \sp }
                                          { \pick\s{\E\op\X}, \pick\sp{\E\opp\X}}              \\
 \E {\op \alt \opp} \X      & = & \concat{\E\op\X}{\E\opp\X}                                   \\
 \E \fail \X                & = & \seq{}                                                       \\
 \E {\scope\op}\X           & = & \seqcomp {\setmin\s\Y} {\pick \s{\E\op\Xp}}
                                  \SPP\where  {\Y = \Exis\op} \and \Xp=\X\cup\Y                \\
 \E {\If\op\opp\oppp}\X     & = & \IF\Y\s{\E\opp\Xp}{\E\opp\X}                                 \\
                            &   & \where{\Y=\Exis\op \and \Xp =\X\cup\Y}                       \\
                            &   & \SPP \s = \ONE\Y{\E\op\Xp}                                   \\
 \E {\x=\For\op\opp\y}\X    & = & \seqcomp{ \setcomp {\setmin\rho{\{\y\}}}
                                                     {\pick\rho\s \and \inrho\x=\inrho\y}}
                                          { \pick\s{\FOR\y\Gen\Bod} }                            \\
                            &   & \where{\ Y = \Exis\op  \and \Xp = \X \cup \Y}                \\
                            &   & \Gen = \E\op\Xp                                              \\
                            &   & \Bod=\seqcomp{\setcomp{\rho[\y\mapsto\tuple{\inrho\y}]}{\rho\in\s}}
                                               {\pick\s{\E\opp{(\Xp\cup\{\y\})}}}
\end{array}

\\ \\

\textbf{Auxiliaries}                                                                           \\
\begin{array}{rcl}
  \IF{\_}{\_}{\_}{\_} & : &  \totfun{\Set\Ident \times \Set\Env \times \Ws \times \Ws}{\Ws}    \\
  \IF\Y\s\Thens\Elses & = &  \seqcomp { \setmin{(\s \cap \sp)}\Y} { \pick\sp\Thens }           \\
                      &   &  \UU                                                              \\
                      &   &  \seqcomp { \sp-(\setmin\s\Y) } { \pick\sp\Elses }                 \\\\
  \ONE{\_}{\_}        & : & \totfun{\Set\Ident \times \Ws}{\Set\Env}                           \\
  \ONE\X{\seq{}}      & = & \set{}                                                             \\
  \ONE\X{\s:\S}       & = & \s \UU \setcomp\rho{\rho\in\ONE\X\S \and \forall\rhop\in\s.\SP
                                                       (\setmin\rho\X)\not=(\setmin\rhop\X)}   \\
                                                                                               \\
  \EMPTY\y            & : &  \totfun\Ident\Ws                                                  \\
  \EMPTY\y            & = &  \seq{ \setcomp{ \inrho\y=\tuple{} } { \pick\rho{\Envs\X} } }        \\
                                                                                               \\
  \APPEND{\_}{\_}{\_} & : &  \totfun{\Ident \times \Ws \times \Ws}\Ws                          \\
  \APPEND\y\S\Sp      & = &  \seqcomp {
                                  \setcomp{ \rho[\y\mapsto\concat{\inrho\y}{\rhop(\y)}] }
                                          { \pick\rho\s \and \pick\rhop\sp \and
                                            \setmin\rho\Y=\setmin\rhop\Y=\rho }
                                   \\&& \SPP } { \pick\s\S \and \pick\sp\Sp }                  \\
                      &   & \where{ \Y = \{\y\}}                                               \\
                                                                                               \\
  \FOR{\_}{\_}{\_}    & : &  \totfun{\Ident \times \Ws \times \Ws}\Ws                          \\
  \FOR\y{[]}{\_}      & = &  \EMPTY\y                                                          \\
  \FOR\y{\s:\S}\Sp    & = &  \APPEND\y { \IF\Y\s\Sp{\EMPTY\y} } { \FOR\y\S\Sp   }                \\
\end{array}

\\ \\

\F{}{} \text{returns a set of functions. For closed lambdas that set has at most one element.}

\\

\begin{array}{rcl}
 \F {\lam q \x\op\opp\y}{}  & : & \totfun \Env {\Set\Fun}                                      \\
\end{array}

  \end{array}
